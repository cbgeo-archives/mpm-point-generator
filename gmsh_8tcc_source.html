200 //! \param[in] ngauss_points Number of gauss points per coordinate
201 template <unsigned Tdim, unsigned Tvertices>
202 void GMSH<Tdim, Tvertices>::compute_material_points(unsigned ngauss_points) {
203

205  ngauss_points_ = ngauss_points;
206  std::vector<double> gauss_constants =
207  element::gauss_points.find(ngauss_points_)->second;
208 
209  //! Create a matrix of xi from gauss points
210  //! Matrix is size npoints x Tdim
211  //! For 3D only
212  unsigned npoints = std::pow(ngauss_points_, Tdim);
213  Eigen::MatrixXd xi_gauss_points(npoints, Tdim);
214  unsigned counter = 0;
215  for (unsigned ii = 0; ii < ngauss_points_; ++ii) {
216  for (unsigned jj = 0; jj < ngauss_points_; ++jj) {
217  for (unsigned kk = 0; kk < ngauss_points_; ++kk) {
218  xi_gauss_points(counter, 0) = gauss_constants.at(ii);
219  xi_gauss_points(counter, 1) = gauss_constants.at(jj);
220  xi_gauss_points(counter, 2) = gauss_constants.at(kk);
221  ++counter;
222  }
223  }
224  }

227 
228  //! last_global_id should be changed later if more than one material
229  //! properties are used
230  //! material_id is the index of materialpoints
231  unsigned last_global_id = 0;
232  unsigned material_id = 0;
233 
234  //! Update vector of material points
235  //! Fill materialpoints_ vector for the first component
236  materialpoints_.emplace_back(std::unique_ptr<MaterialPoints<Tdim>>(
237  new MaterialPoints<Tdim>(material_id)));
238 
239  for (const auto& element : elements_) {
240 
241  //! Store coordinates in Tdim x Tvertices matrix
242  //! Where N is the number of nodes per element
243  //! This is rearranging of the data to have stored in matrix form
244  Eigen::MatrixXd node_coordinates(Tdim, Tvertices);
245  for (unsigned i = 0; i < Tvertices; ++i) {
246  for (unsigned j = 0; j < Tdim; ++j) {
247  node_coordinates(j, i) = element->vertex_coordinates(i)[j];
248  }
249  }
250 
251  //! Compute the gauss points (there are npoints)
252  for (unsigned k = 0; k < npoints; ++k) {
253 
254  //! Get array of xi for this gauss point
255  std::array<double, Tdim> xi;
256  for (unsigned l = 0; l < Tdim; ++l) xi.at(l) = xi_gauss_points(k, l);
257 
258  // Compute gauss point in cartesian coordinate
259  for (unsigned i = 0; i < Tdim; ++i) {
260  pointsarray[i] = 0;
261  Eigen::VectorXd shape_function = element::hexahedron::shapefn(xi);
262  for (unsigned j = 0; j < Tvertices; ++j) {
263  pointsarray[i] += shape_function[j] * node_coordinates(i, j);
264  }
265  }

270  element->id(), element->id() + last_global_id, pointsarray)));
271  }
272  }

275  unsigned nmaterialpoints = 0;
276

278  nmaterialpoints += materialpoint->coordinates().size();
279 
280  std::cout << "Number of Material Points: " << nmaterialpoints << '\n';
281 }
